uvicorn>=0.20,<1.0
Pillow>=10.0,<12
httpx>=0.25,<1.0
numpy>=1.26,<3
pycairo>=1.25,<2

cairosvg>=2.7,<3
# Classic-CV burned-in-text detection for mislabelled "textless" posters.

opencv-python-headless>=4.8,<5
